162   public String toString(CpoClass cpoClass) throws CpoException {
163     StringBuilder sb = new StringBuilder();
164     CpoAttribute cpoAttribute = null;
165 
166 
167     if (getLogical() != CpoWhere.LOGIC_NONE) {
168       sb.append(" ");
169       sb.append(logicals[getLogical()]);
170     } else if (!hasParent()) {
171       // This is the root where clause
172       sb.append("WHERE");
173     }
174 
175     if (getNot()) {
176       sb.append(" NOT");
177     }
178 
179     if (getAttribute() != null) {
180       if (sb.length() > 0) {
181         sb.append(" ");
182       }
183       String fullyQualifiedColumn;
184 
185       cpoAttribute = cpoClass.getAttributeJava(getAttribute());
186       if (cpoAttribute == null) {
187         // This is not an attribute on the cpo bean passed to the retrieveBeans method.
188         // treat it as the column name
189         fullyQualifiedColumn = getAttribute();
190       } else {
191         fullyQualifiedColumn = buildColumnName(cpoAttribute);
192       }
193 
194       if (getAttributeFunction() != null) {
195         if (cpoAttribute != null) {
196           sb.append(buildFunction(getAttributeFunction(), cpoAttribute.getJavaName(), fullyQualifiedColumn));
197         } else {
198           sb.append(getAttributeFunction());
199         }
200       } else {
201         sb.append(fullyQualifiedColumn);
202       }
203     }
204 
205     if (getComparison() != CpoWhere.COMP_NONE) {
206       sb.append(" ");
207       sb.append(comparisons[getComparison()]);
208     }
209 
210     if (getComparison() != CpoWhere.COMP_ISNULL && (getValue() != null || getRightAttribute() != null || getStaticValue() != null)) {
211       sb.append(" ");
212 
213       if (getValue() != null) {
214         if (getValueFunction() != null) {
215           if (cpoAttribute == null) {
216             cpoAttribute = cpoClass.getAttributeJava(getRightAttribute());
217           }
218           sb.append(buildFunction(getValueFunction(), getAttributeName(cpoAttribute, getAttribute(), getRightAttribute()), "?"));
219         } else if (getComparison() == CpoWhere.COMP_IN && getValue() instanceof Collection) {
220           Collection coll = (Collection) getValue();
221           sb.append("(");
222           if (coll.size() > 0) {
223             sb.append("?"); // add the parameter, we will bind it later.
224             for (int i = 1; i < coll.size(); i++) {
225               sb.append(", ?"); // add the parameter, we will bind it later.
226             }
227           }
228           sb.append(")");
229         } else {
230           sb.append("?"); // add the parameter, we will bind it later.
231         }
232       } else if (getRightAttribute() != null) {
233         cpoAttribute = cpoClass.getAttributeJava(getRightAttribute());
234         String fullyQualifiedColumn;
235         if (cpoAttribute == null) {
236           fullyQualifiedColumn = getRightAttribute();
237         } else {
238           fullyQualifiedColumn = buildColumnName(cpoAttribute);
239         }
240 
241         if (getRightAttributeFunction() != null) {
242           sb.append(buildFunction(getRightAttributeFunction(), getAttributeName(cpoAttribute, getAttribute(), getRightAttribute()), fullyQualifiedColumn));
243         } else {
244           sb.append(fullyQualifiedColumn);
245         }
246       } else if (getStaticValue() != null) {
247         sb.append(getStaticValue());
248       }
249     }
250     return sb.toString();
251   }

253   private String getAttributeName(CpoAttribute jdbcAttribute, String leftAttribute, String rightAttribute) {
254     String attrName = null;
255 
256     if (jdbcAttribute != null) {
257       attrName = jdbcAttribute.getJavaName();
258     }
259 
260     if (attrName == null && leftAttribute != null) {
261       attrName = leftAttribute;
262     }
263 
264     if (attrName == null && rightAttribute != null) {
265       attrName = rightAttribute;
266     }
267 
268     return attrName;
269   }

310   private String buildFunction(String function, String match, String value) {
311     StringBuilder sb = new StringBuilder();
312     int attrOffset;
313     int fromIndex = 0;
314 
315     if (function != null && function.length() > 0) {
316       while ((attrOffset = function.indexOf(match, fromIndex)) != -1) {
317         sb.append(function.substring(0, attrOffset));
318         sb.append(value);
319         fromIndex += attrOffset + match.length();
320       }
321       sb.append(function.substring(fromIndex));
322     }
323 
324     return sb.toString();
325   }

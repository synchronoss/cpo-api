27  import java.sql.Timestamp;
28  import java.util.ArrayList;

30  /**
31   * InsertObjectTest is a JUnit test class for testing the insert api calls of cpo
32   *
33   * @author david berry
34   */
35  public class CaseInsensitiveTest extends TestCase {
36  
37    private ArrayList<ValueObject> al = new ArrayList<>();
38    private CpoAdapter cpoAdapter = null;
39    private CpoAdapter readAdapter = null;
40    private JdbcCpoMetaDescriptor metaDescriptor = null;
41  
42    public CaseInsensitiveTest(String name) {
43      super(name);
44    }
45  
46    /**
47     * <code>setUp</code> Load the datasource from the properties in the property file jdbc_en_US.properties
48     *
49     * @author david berry
50     * @version '$Id: InsertObjectTest.java,v 1.3 2006/01/30 19:09:23 dberry Exp $'
51     */
52    @Override
53    public void setUp() {
54      String method = "setUp:";
55  
56      try {
57        cpoAdapter = CpoAdapterFactory.getCpoAdapter(JdbcStatics.ADAPTER_CONTEXT_CASEINSENSITIVE);
58        assertNotNull(method + "IdoAdapter is null", cpoAdapter);
59        metaDescriptor = (JdbcCpoMetaDescriptor) cpoAdapter.getCpoMetaDescriptor();
60      } catch (Exception e) {
61        fail(method + e.getMessage());
62      }
63      try {
64        readAdapter = CpoAdapterFactory.getCpoAdapter(JdbcStatics.ADAPTER_CONTEXT_CASEINSENSITIVE);
65        assertNotNull(method + "IdoAdapter is null", readAdapter);
66      } catch (Exception e) {
67        fail(method + e.getMessage());
68      }
69    }
70  
71    public void testCaseInsensitiveObject() {
72      String method = "testCaseInsensitiveObject:";
73    
74      ValueObject valObj = new ValueObject(5);
75  
76      valObj.setAttrVarChar("testCaseInsensitiveObject");
77      valObj.setAttrInteger(3);
78      Timestamp ts = new Timestamp(System.currentTimeMillis());
79  
80      if (!metaDescriptor.isSupportsMillis()) {
81        ts.setNanos(0);
82      }
83  
84      valObj.setAttrDatetime(ts);
85  
86      valObj.setAttrBit(true);
87  
88      al.add(valObj);
89  
90      try {
91        cpoAdapter.insertObject(valObj);
92      } catch (Exception e) {
93        fail(method + e.getMessage());
94      }
95  
96      try {
97        ValueObject vo = readAdapter.retrieveBean(null, valObj, valObj, null, null);
98        assertTrue("Ids do not match", vo.getId() == valObj.getId());
99        assertTrue("Integers do not match", vo.getAttrInteger() == valObj.getAttrInteger());
100       assertEquals("Strings do not match", vo.getAttrVarChar(), valObj.getAttrVarChar());
101       assertEquals("Timestamps do not match", vo.getAttrDatetime(), valObj.getAttrDatetime());
102       assertTrue("boolean not stored correctly", vo.getAttrBit());
103 
104     } catch (Exception e) {
105       fail(method + e.getMessage());
106     }
107 
108 
109   }
110

112   public void tearDown() {
113     String method = "tearDown:";
114     try {
115       cpoAdapter.deleteObjects(al);
116 
117     } catch (Exception e) {
118       fail(method + e.getMessage());
119     }
120     cpoAdapter = null;
121     readAdapter = null;
122   }

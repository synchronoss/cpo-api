25  /**
26   * BindableCpoOrderBy is an interface for specifying the sort order in which objects are returned from the Datasource.
27   *
28   * @author david berry
29   */
30  public class BindableCpoOrderBy implements CpoOrderBy {
31  
32    private boolean ascending;
33    private String attribute;
34    private String function;
35    private String marker = DEFAULT_MARKER;
36  
37    @SuppressWarnings("unused")
38    private BindableCpoOrderBy() {
39    }
40  
41    public BindableCpoOrderBy(String attr, boolean asc) {
42      ascending = asc;
43      attribute = attr;
44      function = null;
45    }
46  
47    public BindableCpoOrderBy(String marker, String attr, boolean asc) {
48      this.marker = marker;
49      ascending = asc;
50      attribute = attr;
51      function = null;
52    }
53  
54    public BindableCpoOrderBy(String attr, boolean asc, String func) {
55      ascending = asc;
56      attribute = attr;
57      function = func;
58    }
59  
60    public BindableCpoOrderBy(String marker, String attr, boolean asc, String func) {
61      this.marker = marker;
62      ascending = asc;
63      attribute = attr;
64      function = func;
65    }
66  
67    /**
68     * Gets the boolean that determines if the objects will be returned from from the CpoAdapter in Ascending order or
69     * Descending order
70     *
71     * @return boolean true if it is to sort in Ascensing Order false if it is to be sorted in Descending Order
72     */
73    @Override
74    public boolean getAscending() {
75      return this.ascending;
76    }
77  
78    /**
79     * Gets the name of the attribute that is to be used to sort the results from the CpoAdapter.
80     *
81     * @return String The name of the attribute
82     */
83    @Override
84    public String getAttribute() {
85      return this.attribute;
86    }
87  
88    /**
89     * Gets a string representing a datasource specific function call that must be applied to the attribute that will be
90     * used for sorting.
91     *
92     * i.e. - "upper(attribute_name)"
93     *
94     * @return String The name of the function
95     */
96    @Override
97    public String getFunction() {
98      return this.function;
99    }
100 
101   @Override
102   public String toString(CpoClass cpoClass) throws CpoException {
103     StringBuilder sb = new StringBuilder();
104     String column;
105     int attrOffset;
106     int fromIndex = 0;
107 
108     if (attribute != null && attribute.length() > 0) {
109       CpoAttribute jdbcAttribute = cpoClass.getAttributeJava(attribute);
110       if (jdbcAttribute == null) {
111         throw new CpoException(attribute);
112       }
113       sb.append(" ");
114 
115       column = jdbcAttribute.getDataName();
116       if (column != null && column.length() > 0) {
117         if (function != null && function.length() > 0) {
118           while ((attrOffset = function.indexOf(attribute, fromIndex)) != -1) {
119             sb.append(function.substring(0, attrOffset));
120             sb.append(column);
121             fromIndex += attrOffset + attribute.length();
122           }
123           sb.append(function.substring(fromIndex));
124         } else {
125           sb.append(column);
126         }
127       }
128 
129       if (this.getAscending()) {
130         sb.append(" ASC");
131       } else {
132         sb.append(" DESC");
133       }
134     }
135 
136     return sb.toString();
137   }

139   @Override

141     return marker;

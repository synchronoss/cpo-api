21  package org.synchronoss.cpo.jdbc;
22  
23  import junit.framework.TestCase;
24  import org.slf4j.*;
25  import org.synchronoss.cpo.*;
26  import org.synchronoss.cpo.helper.ExceptionHelper;
27  
28  import java.util.ArrayList;
29  
30  /**
31   * BigBatchTest is a JUnit test class for testing big batches
32   *
33   * @author david berry
34   */
35  public class BigBatchTest extends TestCase {
36  
37    private static final Logger logger = LoggerFactory.getLogger(BigBatchTest.class);
38    private ArrayList<ValueObject> al = new ArrayList<>();
39    private CpoAdapter cpoAdapter = null;
40  
41    public BigBatchTest(String name) {
42      super(name);
43    }
44  
45    /**
46     * <code>setUp</code> Load the datasource from the properties in the property file jdbc_en_US.properties
47     *
48     * @author david berry
49     * @version '$Id: InsertObjectTest.java,v 1.3 2006/01/30 19:09:23 dberry Exp $'
50     */
51    @Override
52    public void setUp() {
53      String method = "setUp:";
54  
55      try {
56        cpoAdapter = CpoAdapterFactory.getCpoAdapter(JdbcStatics.ADAPTER_CONTEXT_JDBC);
57        assertNotNull(method + "IdoAdapter is null", cpoAdapter);
58      } catch (Exception e) {
59        fail(method + e.getMessage());
60      }
61    }
62  
63    /**
64     * So oracle seems to fail on a batch size of 100,000 but does not throw an error.
65     *
66     * lets try to break it to fix it to return a good message.
67     *
68     */
69    public void testTooManyInserts() {
70  
71      String method = "testTooManyInserts:";
72      int numInserts = 100000;
73  
74      for (int i = 0; i < numInserts; i++) {
75        al.add(new ValueObjectBean(i));
76      }
77  
78      try {
79        long inserts = cpoAdapter.insertObjects(al);
80        assertEquals("inserts performed do not equal inserts requested: ", inserts, numInserts);
81      } catch (CpoException ce) {
82        logger.debug("Received a CpoException:" + ExceptionHelper.getLocalizedMessage(ce));
83      } catch (Exception e) {
84        logger.error(ExceptionHelper.getLocalizedMessage(e));
85        fail(method + ":Received an Exception instead of a CpoException: " + ExceptionHelper.getLocalizedMessage(e));
86      } catch (Throwable t) {
87        logger.error(ExceptionHelper.getLocalizedMessage(t));
88        fail(method + ":Received a Throwable instead of a CpoException: " + ExceptionHelper.getLocalizedMessage(t));
89      }
90  
91    }
92  
93    @Override
94    public void tearDown() {
95      String method = "tearDown:";
96      try {
97        cpoAdapter.deleteObjects(al);
98  
99      } catch (Exception e) {
100             logger.error(ExceptionHelper.getLocalizedMessage(e));
101       fail(method + e.getMessage());
102     }
103     cpoAdapter = null;
104   }
105 }

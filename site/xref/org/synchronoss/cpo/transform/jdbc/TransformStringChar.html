21  package org.synchronoss.cpo.transform.jdbc;
22  
23  import org.synchronoss.cpo.CpoException;
24  import org.synchronoss.cpo.jdbc.JdbcCallableStatementFactory;
25  import org.synchronoss.cpo.jdbc.JdbcPreparedStatementFactory;
26  
27  /**
28   * Converts a java.sql.Blob from a jdbc datasource to a byte[] and from a byte[] to a java.sql.Blob
29   *
30   * @author david berry
31   */
32  public class TransformStringChar implements JdbcCpoTransform<String, char[]> {
33  
34    public TransformStringChar() {
35    }
36  
37    /**
38     * Transforms the datasource object into an object required by the class
39     *
40     * @param cpoAdapter The CpoAdapter for the datasource where the attribute is being retrieved
41     * @param parentObject The object that contains the attribute being retrieved.
42     * @param The object that represents the datasource object being retrieved
43     * @return The object to be stored in the attribute
44     * @throws CpoException
45     */
46    @Override
47    public char[] transformIn(String lvc) throws CpoException {
48  
49      return lvc == null ? null : lvc.toCharArray();
50    }
51  
52    /**
53     * Transforms the data from the class attribute to the object required by the datasource
54     *
55     * @param cpoAdapter The CpoAdapter for the datasource where the attribute is being persisted
56     * @param parentObject The object that contains the attribute being persisted.
57     * @param attributeObject The object that represents the attribute being persisted.
58     * @return The object to be stored in the datasource
59     * @throws CpoException
60     */
61    @Override
62    public String transformOut(JdbcPreparedStatementFactory jpsf, char[] attrObj) throws CpoException {
63      return attrObj == null ? null : String.valueOf(attrObj);
64    }
65  
66    @Override
67    public String transformOut(JdbcCallableStatementFactory jpsf, char[] attributeObject) throws CpoException, UnsupportedOperationException {
68      throw new UnsupportedOperationException("Not supported yet.");
69    }
70  
71    @Override
72    public String transformOut(char[] j) throws CpoException, UnsupportedOperationException {
73      throw new UnsupportedOperationException("Not supported yet.");
74    }
75  }
